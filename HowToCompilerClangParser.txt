Requisitos para compilar um parser

Cmake:

cmake_minimum_required(VERSION 3.20)

project(MeuParser LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(LLVM REQUIRED CONFIG)
find_package(Clang REQUIRED CONFIG)

message(STATUS "LLVM Version: ${LLVM_PACKAGE_VERSION}")

list(APPEND CMAKE_MODULE_PATH "${LLVM_CMAKE_DIR}")
include(HandleLLVMOptions)

llvm_map_components_to_libnames(LLVM_LIBS
    Support
    Core
)

add_executable(MeuParser
    main.cpp
)

target_include_directories(MeuParser PRIVATE
    ${LLVM_INCLUDE_DIRS}
    ${CLANG_INCLUDE_DIRS}
)

target_compile_definitions(MeuParser PRIVATE
    ${LLVM_DEFINITIONS}
)

target_link_libraries(MeuParser PRIVATE
    ${LLVM_LIBS}

    clangTooling
    clangToolingCore

    clangFrontend
    clangFrontendTool

    clangDriver
    clangSerialization

    clangAST
    clangASTMatchers

    clangBasic
    clangLex
    clangParse
    clangSema

    clangEdit

    clangAnalysis

    clangRewrite
    clangRewriteFrontend

    clangIndex
)





gerar build:

cmake -S . -B build -G "Visual Studio 18 2026" -A x64 -DLLVM_DIR="C:\LLVM22\lib\cmake\llvm" -DClang_DIR="C:\LLVM22\lib\cmake\clang"

compilar:

cmake --build build --config Release
